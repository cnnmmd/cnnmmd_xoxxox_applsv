FROM ubuntu:24.04

ENV DEBIAN_FRONTEND=noninteractive

RUN mkdir ${PTHEXP}

  apt-get install -y --no-install-recommends pkg-config ninja-build libopenblas-dev && \
  rm -rf /var/lib/apt/lists/*
